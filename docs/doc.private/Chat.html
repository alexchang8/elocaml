Module Chat

module Chat: sig .. end

The module that holds chat logic

type t = string list

The abstract type representing a chat state

val n_list : 'a -> int -> 'a list

n_list x n creates a list of n elements, where every element is x

val extend_string : string -> string

extend_string s returns s if s has 25 or more characters, or otherwise adds whitespace at the end of s until it has exactly 25 characters.

val next_state : string -> t -> t

next_state s t returns the state of the chat after s is inserted into the chat

val list_take_n : 'a list -> int -> 'a list

list_take_n lst n Returns the first n elements of lst. If there are less than n elements in lst, then it returns lst

val print_chat : t -> string

print_chat t returns a string that represents a properly formatted view of chat
