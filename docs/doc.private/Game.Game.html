Module type Game.Game

module type Game = sig .. end
type t

The abstract type representing game state

type command

The abstract type representing a parsed player input

val init_state : t

init_state is an abstract value representing the initial state of a game

val max_players : int

max_players is the number of clients that must connect to the server for a game to start

val next_state : t -> int -> command -> t

next_state t p_id command returns the abstract state of the game after a player with p_id inputs a string corresponding to command.

val parse : string -> command

parse s returns the command corresponding to s. The implementation depends on the game

val print_player_state : t -> int -> string

print_player_state t id is a string representing the perspective of the player corresponding to id

val name : string

String that is the title of the game

val terminal_size : int * int
